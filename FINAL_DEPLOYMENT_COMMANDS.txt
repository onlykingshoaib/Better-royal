# Download from: https://devcenter.heroku.com/articles/heroku-cli

# Step 2: Run these commands

heroku create your-app-name
git push heroku main
heroku open

# Step 3: View logs (if needed)
heroku logs --tail

Result: https://your-app-name.herokuapp.com

METHOD 3: DOCKER (FOR YOUR OWN SERVER)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

# Build Docker image
cd /workspaces/Better-royal
docker build -t better-royal .

# Run locally to test
docker run -p 8000:5000 better-royal

# Deploy to cloud (AWS, Google Cloud, etc.)
docker tag better-royal your-registry/better-royal
docker push your-registry/better-royal

Result: http://your-server-ip:8000
